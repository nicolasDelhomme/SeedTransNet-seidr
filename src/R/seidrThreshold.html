th2 <- melt(th, id = "Threshold")
ggplot(th2, aes(x = Threshold, y = value, group = variable, col = variable)) +
  geom_line(lwd = 0.5) + facet_wrap(~variable, scales = "free") +
  scale_x_reverse() + theme_bw() +
